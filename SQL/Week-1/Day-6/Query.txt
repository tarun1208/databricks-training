1.
select e.emp_name,d.dept_name,e.salary
from employees e 
join departments d
on e.dept_id = d.dept_id

2.
select d.dept_name,count(e.emp_id) as employee_count
from departments d
left join employees e
on d.dept_id = e.dept_id
group by d.dept_id,d.dept_name

3.
select e.emp_id,e.emp_name, e1.emp_name as manager_name
from employees e
left join employees e1
on e.manager_id = e1.emp_id

4.
select d.dept_name,sum(e.salary)as total_salary
from departments d
left join employees e
on d.dept_id = e.dept_id
group by d.dept_id,d.dept_name

5.
select d.dept_name,avg(e.salary)as avg_salary
from departments d
left join employees e
on d.dept_id = e.dept_id
group by d.dept_id,d.dept_name

6.
select d.dept_name,avg(e.salary)as avg_salary
from departments d
left join employees e
on d.dept_id = e.dept_id
group by d.dept_id,d.dept_name
order by avg_salary desc
limit 1

7.
select emp_id,emp_name,salary
from employees
where salary>(select Avg(salary) from employees)

8.
select emp_id,emp_name,dept_id,salary
from employees e
where salary >(
    select avg(salary)
    from employees
    where dept_id = e.dept_id)

9.
select salary
from employees
where salary <(
  select max(salary)
  from employees)
order by salary desc
limit 1

10.
select emp_id,emp_name,dept_id,salary
from employees e
where salary =(
    select max(salary)
    from employees
    where dept_id = e.dept_id)

11.
select emp_id,emp_name,dept_id,salary,
rank() over( partition by dept_id order by salary desc)as salary_rank
from employees

12.
select *
from (
	select emp_id,emp_name,dept_id,salary,
	rank() over( partition by dept_id order by salary desc)as salary_rank
	from employees
  )x
where salary_rank <=3

13.
select emp_id,emp_name,hire_date,salary,
sum(salary) over(order by hire_date)as running_total_salary
from employees

14.
select emp_id,emp_name,salary,
LAG(salary) over (order by salary)as previous_salary
from employees

15.
select c.customer_id,c.customer_name
from customers c
left join orders o
on c.customer_id = o.customer_id
where c.customer_id not in (select customer_id from orders)

16.
select customer_id,order_id
from orders o
where order_date = (select max(order_date)
                    from orders
                    where customer_id = o.customer_id)

17.
select customer_id,sum(amount)
from orders
group by customer_id

18.
select customer_id,sum(amount)*100/(select sum(amount) from orders) as customer_cuntribution
from orders
group by customer_id

19.
select customer_name
from customers
group by customer_name
having count(*)>1

20.
select emp_name,salary,
case when salary <80000 then 'Low'
	 when salary between 80000 and 120000 then 'Medium'
     else 'High'
end as salary_brand
from employees

21.
select emp_id,emp_name,salary
from employees
where salary in(
  select salary 
  from employees
  group by salary
  having count(*)>1
  )

22.
select c.customer_id,c.customer_name
from customers c
left join orders o
on c.customer_id = o.customer_id
group by c.customer_id, c.customer_name
having count(*)>3

23.
select month(order_date) as month_num,sum(amount)
from orders
group by month(order_date)
order by sum(amount) desc
limit 1

24.
select customer_id,order_date,
lag(order_date) over(partition by customer_id order by order_date) as previous_date,
datediff(order_date,lag(order_date) over(partition by customer_id order by order_date)
         )as days_differnce
from orders
